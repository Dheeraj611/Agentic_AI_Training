import asyncio

async def worker():
    try:
        while True:
            print("working")
            await asyncio.sleep(1)
    except asyncio.CancelledError:
        print("cancelled")
        raise

async def main():
    task = asyncio.create_task(worker())

    await asyncio.sleep(2.5)

    task.cancel()

    try:
        await task
    except asyncio.CancelledError:
        print("main noticed")

asyncio.run(main())
